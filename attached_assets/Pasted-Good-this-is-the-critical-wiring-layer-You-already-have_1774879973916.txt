🚀 1. BACKEND: CREATE JOB + TRIGGER DISPATCH
/server/routes/jobs.ts
import express from "express"
import { db } from "../db/client"
import { jobs } from "../db/schema"
import { dispatchJob } from "../services/dispatchEngine"

const router = express.Router()

router.post("/create", async (req, res) => {
  try {
    const {
      tier,
      weight,
      items,
      basePrice,
      totalPrice
    } = req.body

    // 🔥 build job requirements from tier
    const tierMap = {
      light: { movers: 1, driver: false, truck: false },
      medium: { movers: 2, driver: true, truck: true },
      large: { movers: 2, driver: true, truck: true },
      xlarge: { movers: 3, driver: true, truck: true }
    }

    const config = tierMap[tier]

    // 🧠 create job
    const [job] = await db.insert(jobs).values({
      serviceType: "junk",
      requiredMovers: config.movers,
      requiresDriver: config.driver,
      requiresTruck: config.truck,
      status: "open",
      estimatedHours: 2
    }).returning()

    // 🚀 trigger dispatch
    await dispatchJob(job.id)

    res.json({
      success: true,
      jobId: job.id
    })

  } catch (err) {
    console.error(err)
    res.status(500).json({ error: "Job creation failed" })
  }
})

export default router
⚙️ 2. DISPATCH ENGINE (HOOKED)
/server/services/dispatchEngine.ts
import { getAvailableCrew } from "./crewEngine"
import { assembleCrew } from "./crewEngine"
import { db } from "../db/client"
import { jobAssignments } from "../db/schema"

export async function dispatchJob(jobId: string) {
  const job = await getJob(jobId)

  const pool = await getAvailableCrew(job)

  const result = assembleCrew(pool, job)

  if (result.status === "fail") {
    console.log("No crew available")
    return
  }

  // assign crew immediately (MVP version)
  for (const member of result.crew) {
    await db.insert(jobAssignments).values({
      jobId,
      contractorId: member.id,
      role: member.role
    })
  }

  console.log("Crew assigned:", result.crew.length)
}

🔄 4. REAL-TIME STATUS (POLLING)
/components/JobStatus.tsx
"use client"

import { useEffect, useState } from "react"

export default function JobStatus({ jobId }) {
  const [status, setStatus] = useState("searching")

  useEffect(() => {
    const interval = setInterval(async () => {
      const res = await fetch(`/api/jobs/${jobId}`)
      const data = await res.json()

      setStatus(data.status)
    }, 3000)

    return () => clearInterval(interval)
  }, [])

  return (
    <div className="p-4 text-center">
      {status === "open" && "🔍 Finding crew..."}
      {status === "assigned" && "🟢 Crew Assigned!"}
    </div>
  )
}
