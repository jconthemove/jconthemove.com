import { MapContainer, TileLayer, Marker, Popup } from "react-leaflet";
import { useEffect, useState } from "react";
import "leaflet/dist/leaflet.css";

export default function AdminMap() {
  const [jobs, setJobs] = useState<any[]>([]);
  const [crew, setCrew] = useState<any[]>([]);

  const fetchData = async () => {
    const j = await fetch("/api/admin/jobs").then(r => r.json());
    const c = await fetch("/api/admin/crew").then(r => r.json());
    setJobs(j);
    setCrew(c);
  };

  useEffect(() => {
    fetchData();
    const i = setInterval(fetchData, 3000);
    return () => clearInterval(i);
  }, []);

  return (
    <div className="h-screen w-full">
      <MapContainer center={[46.454, -90.172]} zoom={12} className="h-full">
        <TileLayer
          attribution="&copy; OpenStreetMap"
          url="https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png"
        />

        {/* JOBS */}
        {jobs.map(job => (
          job.lat && job.lng && (
            <Marker key={`job-${job.id}`} position={[job.lat, job.lng]}>
              <Popup>
                Job #{job.id}<br/>
                {job.services}<br/>
                ${job.total}
              </Popup>
            </Marker>
          )
        ))}

        {/* CREW */}
        {crew.map(c => (
          c.lat && c.lng && (
            <Marker key={`crew-${c.id}`} position={[c.lat, c.lng]}>
              <Popup>
                {c.name}<br/>
                Jobs Today: {c.jobsToday}
              </Popup>
            </Marker>
          )
        ))}
      </MapContainer>
    </div>
  );
}

📁 server/geo.ts
export function haversine(a: any, b: any) {
  const R = 6371;
  const dLat = (b.lat - a.lat) * Math.PI / 180;
  const dLon = (b.lng - a.lng) * Math.PI / 180;

  const lat1 = a.lat * Math.PI / 180;
  const lat2 = b.lat * Math.PI / 180;

  const x =
    Math.sin(dLat / 2) ** 2 +
    Math.sin(dLon / 2) ** 2 * Math.cos(lat1) * Math.cos(lat2);

  return R * 2 * Math.atan2(Math.sqrt(x), Math.sqrt(1 - x));
}
📁 server/autoDispatch.ts (UPGRADE)
import { haversine } from "./geo";

export function autoAssignCrew(job: any, crewList: any[]) {
  const available = crewList.filter(c => c.online && c.lat && c.lng);

  if (!job.lat || !job.lng) return null;

  const scored = available.map(c => {
    const distance = haversine(
      { lat: job.lat, lng: job.lng },
      { lat: c.lat, lng: c.lng }
    );

    let score = 100;

    // CLOSEST CREW WINS
    score -= distance * 10;

    // LOAD BALANCE
    score -= c.jobsToday * 5;

    return { ...c, score };
  });

  scored.sort((a, b) => b.score - a.score);

  return scored[0];
}
