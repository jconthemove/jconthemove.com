export function calculateRewards(total: number) {
  const rate = 15; // 15 JCMOVES per $1 (adjust later)

  const base = total * rate;

  let tierBonus = 1;

  if (total > 1000) tierBonus = 1.2;
  if (total > 2500) tierBonus = 1.4;

  const final = Math.round(base * tierBonus);

  return {
    base,
    bonusMultiplier: tierBonus,
    final,
  };
}

export function assignCrew(job: any, crewList: any[]) {
  const requiredMovers =
    job.services.includes("moving") ? 2 : 1;

  const available = crewList
    .filter((c) => c.online)
    .sort((a, b) => a.jobsToday - b.jobsToday);

  return available.slice(0, requiredMovers);
}

export function applyDynamicPricing(total: number, context: any) {
  let multiplier = 1;

  // High demand (weekends)
  if (context.day === "Saturday" || context.day === "Sunday") {
    multiplier += 0.1;
  }

  // Low crew availability
  if (context.availableCrew < 2) {
    multiplier += 0.15;
  }

  // Slow day discount
  if (context.jobsToday < 3) {
    multiplier -= 0.1;
  }

  return {
    adjustedTotal: Math.round(total * multiplier),
    multiplier,
  };
}
