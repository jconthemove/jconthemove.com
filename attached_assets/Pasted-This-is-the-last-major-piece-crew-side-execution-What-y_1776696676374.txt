import { useEffect, useState } from "react";

export default function CrewApp() {
  const [jobs, setJobs] = useState<any[]>([]);
  const [activeJob, setActiveJob] = useState<any>(null);
  const [earnings, setEarnings] = useState(0);

  const fetchJobs = async () => {
    const res = await fetch("/api/crew/jobs");
    const data = await res.json();
    setJobs(data);
  };

  useEffect(() => {
    fetchJobs();
    const interval = setInterval(fetchJobs, 3000);
    return () => clearInterval(interval);
  }, []);

    await fetch("/api/crew/accept", {
      method: "POST",
      body: JSON.stringify({ jobId }),
    });
    fetchJobs();
  };

  const declineJob = async (jobId: number) => {
    await fetch("/api/crew/decline", {
      method: "POST",
      body: JSON.stringify({ jobId }),
    });
  };

  const updateStatus = async (status: string) => {
    await fetch("/api/crew/status", {
      method: "POST",
      body: JSON.stringify({ jobId: activeJob.id, status }),
    });
    fetchJobs();
  };

  return (
    <div className="p-4 text-white bg-black min-h-screen">
      <h1 className="text-xl font-bold mb-4">Crew App</h1>

      {/* ACTIVE JOB */}
      {activeJob && (
        <div className="bg-green-900 p-4 mb-4 rounded">
          <p>Active Job #{activeJob.id}</p>
          <p>{activeJob.services}</p>
          <p>Bonus: ${activeJob.crewBonus}</p>

          <a
            href={`https://www.google.com/maps/dir/?api=1&destination=${activeJob.lat},${activeJob.lng}`}
            target="_blank"
            className="block bg-blue-500 p-2 mt-2 text-center rounded"
          >
            Navigate
          </a>

          <button onClick={() => updateStatus("en_route")}>En Route</button>
          <button onClick={() => updateStatus("on_site")}>On Site</button>
          <button onClick={() => updateStatus("completed")}>Complete</button>
        </div>
      )}

      <h2 className="text-lg mb-2">Available Jobs</h2>

      {jobs.map((job) => (

          <p>Job #{job.id}</p>
          <p>{job.services}</p>
          <p>${job.total}</p>
          <p>Bonus: ${job.crewBonus}</p>

🔌 2. CREW API ROUTES
📁 server/routes/crew.ts
import express from "express";
import { db } from "../db";
import { jobs } from "../db/schema";

const router = express.Router();

// GET AVAILABLE JOBS
router.get("/crew/jobs", async (req, res) => {
  const available = await db
    .select()
    .from(jobs)
    .where(jobs.status.eq("assigned"));

  res.json(available);
});

// ACCEPT JOB
router.post("/crew/accept", async (req, res) => {
  const { jobId } = req.body;

  await db.update(jobs)
    .set({ status: "accepted" })
    .where(jobs.id.eq(jobId));

  res.json({ success: true });
});

// DECLINE JOB
router.post("/crew/decline", async (req, res) => {
  const { jobId } = req.body;

  await db.update(jobs)
    .set({ status: "open" })
    .where(jobs.id.eq(jobId));

  res.json({ success: true });
});

// UPDATE STATUS
router.post("/crew/status", async (req, res) => {
  const { jobId, status } = req.body;

  await db.update(jobs)
    .set({ status })
    .where(jobs.id.eq(jobId));

  res.json({ success: true });
});

export default router;
