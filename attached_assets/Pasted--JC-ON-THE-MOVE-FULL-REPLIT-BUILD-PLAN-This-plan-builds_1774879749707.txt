    if (!matchesJobRequirements(c, job)) return false
    if (!(await canTakeJob(c.id, job))) return false
    return true
  })
}
🚛 crew auto-assembly (CRITICAL)
export function assembleCrew(pool, job) {
  const req = buildRequirements(job)

  let crew = []

  const driver = pickBestDriver(pool, req)
  if (!driver) return { status: "fail" }

  crew.push(driver)

  const movers = pickMovers(pool, req.movers, crew)

  if (movers.length < req.movers) {
    return {
      status: "partial",
      crew,
      missing: req.movers - movers.length
    }
  }

  return {
    status: "complete",
    crew: [...crew, ...movers]
  }
}
📡 dispatchEngine.ts
export async function dispatchJob(jobId) {
  const job = await getJob(jobId)

  const pool = await getAvailableCrew(job)

  const crew = assembleCrew(pool, job)

  if (crew.status === "fail") {
    triggerDemandBoost()
    return
  }

  broadcastToCrew(job, crew)
}

pricingEngine.ts
export function calculatePrice(job) {
  const tier = resolveJunkTier(job.weight)

  let total = JUNK_TIERS[tier].price

  for (const item of job.items) {
    total += item.price * item.quantity
  }

  return total * getDemandMultiplier()
}
