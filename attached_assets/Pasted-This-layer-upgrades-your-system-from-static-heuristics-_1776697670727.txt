Zone Mapping
📁 zones.ts
import { haversine } from "../geo";

export function getZone(lat: number, lng: number) {
  for (const z of ZONES) {
    const d = haversine(z.center, { lat, lng });
    if (d <= z.radiusMiles) return z.id;
  }
  return "out_of_zone";
}
⏱️ ROLLING WINDOWS
📁 server/demand/windows.ts
export function buildWindows(jobs) {
  const now = Date.now();

  return jobs.reduce((acc, j) => {
    const diff = now - j.createdAt;

    if (diff < 15 * 60_000) acc.last15m++;
    if (diff < 60 * 60_000) acc.last60m++;
    if (diff < 24 * 60 * 60_000) acc.last24h++;

    return acc;
  }, {
    last15m: 0,
    last60m: 0,
    last24h: 0
  });
}
📊 DEMAND SCORING
📁 server/demand/scorer.ts
export function scoreDemand({ windows, activeJobs, crewAvailable, avgValue }) {
  let score = 0;

  // Short-term spike
  score += windows.last15m * 0.4;

  // Mid-term trend
  score += windows.last60m * 0.3;

  // Baseline
  score += windows.last24h * 0.1;

  // Active workload
  score += activeJobs * 0.2;

  // Crew shortage multiplier
  if (crewAvailable <= 1) score *= 1.5;

  // Normalize (tune later)
  return Math.min(score / 10, 1.5);
}
💰 SURGE PRICING
📁 server/demand/surge.ts
export function computeSurge(base, demandScore, crewAvailable) {
  let multiplier = 1;

  if (demandScore > 0.7) multiplier += 0.15;
  if (demandScore > 1.0) multiplier += 0.3;

  if (crewAvailable === 0) multiplier += 0.5;

  if (demandScore < 0.3 && crewAvailable > 4) {
    multiplier -= 0.1;
  }

  return {
    total: Math.round(base * multiplier),
    multiplier
  };
}
🧭 CREW POSITIONING
📁 server/demand/positioning.ts
export function getPositionSuggestion(zoneScores, crew) {
  if (!crew.online) return null;

  const best = Object.entries(zoneScores)
    .sort((a, b) => b[1] - a[1])[0];

  if (!best) return null;

  return {
    targetZone: best[0],
    reason: "High demand"
  };
}

In demand.step.ts
import { buildWindows } from "../demand/windows";
import { scoreDemand } from "../demand/scorer";
import { getZone } from "../demand/zones";

export async function demandStep(ctx) {
  const jobs = await db.select().from(jobsTable);

  const zone = getZone(ctx.input.lat, ctx.input.lng);

  const zoneJobs = jobs.filter(j => getZone(j.lat, j.lng) === zone);

  const windows = buildWindows(zoneJobs);

  const activeJobs = zoneJobs.filter(j => j.status !== "completed").length;

  const crewAvailable = ctx.crewAvailable ?? 0;

  const avgValue = zoneJobs.reduce((s, j) => s + j.total, 0) / (zoneJobs.length || 1);

  const score = scoreDemand({
    windows,
    activeJobs,
    crewAvailable,
    avgValue
  });

  ctx.demandScore = score;
  ctx.zone = zone;
}

📁 server/demand/metrics.ts
export function buildZoneMetrics(jobs) {
  const zones: Record<string, any> = {};

  for (const j of jobs) {
    const z = getZone(j.lat, j.lng);
    zones[z] ??= { count: 0, revenue: 0 };

    zones[z].count++;
    zones[z].revenue += j.total;
  }

  return zones;
}
