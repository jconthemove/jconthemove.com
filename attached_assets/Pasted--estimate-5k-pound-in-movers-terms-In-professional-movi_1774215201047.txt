const BASE_WEIGHT = {
  studio: 1500,
  "1_bed": 3000,
  "2_bed": 5000,
  "3_bed": 8000,
  "4_plus": 12000
}

function estimateWeight(input: JobInput) {
  let weight = BASE_WEIGHT[input.homeSize]

  weight += input.heavyItems * 500

  if (input.packing) weight *= 1.15
  if (input.longCarry) weight *= 1.10
  if (input.stairs) weight *= 1.10

  return Math.round(weight)
}

5️⃣ PRICING ENGINE
Base rates (YOU control margins here)
const HOURLY_RATE = {
  2: 140,  // 2 movers
  3: 180   // 3 movers
}

function calculatePrice(weight: number, input: JobInput) {
  const crew = getCrew(weight)
  let base = crew.hours * HOURLY_RATE[crew.movers]

  // modifiers
  if (input.stairs) base += 75
  if (input.longCarry) base += 50
  if (input.heavyItems > 0) base += input.heavyItems * 100

  const mileage = input.distanceMiles * 2

  return Math.round(base + mileage)
}

/server/routes/quote.js
import express from "express"

const router = express.Router()

/* ---------------- WEIGHT ENGINE ---------------- */

const BASE_WEIGHT = {
  studio: 1500,
  "1_bed": 3000,
  "2_bed": 5000,
  "3_bed": 8000,
  "4_plus": 12000
}

function estimateWeight(input) {
  let weight = BASE_WEIGHT[input.homeSize] || 3000

  weight += (input.heavyItems || 0) * 500

  if (input.packing) weight *= 1.15
  if (input.longCarry) weight *= 1.10
  if (input.stairs) weight *= 1.10

  // 🔥 protection buffer
  weight *= 1.10

  return Math.round(weight)
}

/* ---------------- CORE LOGIC ---------------- */

function getTier(weight) {
  if (weight < 2000) return "TIER_1"
  if (weight <= 6000) return "TIER_2"
  return "TIER_3"
}

function getCrew(weight) {
  if (weight < 2000) return { movers: 2, hours: 2 }
  if (weight <= 6000) return { movers: 2, hours: 5 }
  return { movers: 3, hours: 8 }
}

const HOURLY_RATE = {
  2: 140,
  3: 180
}

function calculatePrice(weight, input) {
  const crew = getCrew(weight)

  let base = crew.hours * HOURLY_RATE[crew.movers]

  if (input.stairs) base += 75
  if (input.longCarry) base += 50
  if ((input.heavyItems || 0) > 0) {
    base += input.heavyItems * 100
  }

  const mileage = (input.distanceMiles || 0) * 2

  return Math.round(base + mileage)
}

function calculateRewards(price) {
  return Math.round(price * 0.05)
}

/* ---------------- ROUTE ---------------- */

router.post("/quote", async (req, res) => {
  try {
    const input = req.body

    const weight = estimateWeight(input)
    const tier = getTier(weight)
    const crew = getCrew(weight)
    const price = calculatePrice(weight, input)
    const rewards = calculateRewards(price)

    return res.json({
      success: true,
      data: {
        weight,
        tier,
        crew,
        price,
        rewards
      }
    })
  } catch (err) {
    console.error(err)
    res.status(500).json({ success: false })
  }
})

export default router

import { useState } from "react"

export default function QuoteCalculator() {
  const [form, setForm] = useState({
    homeSize: "2_bed",
    heavyItems: 0,
    stairs: false,
    longCarry: false,
    packing: false,
    distanceMiles: 10
  })

  const [result, setResult] = useState<any>(null)

  const handleChange = (key: string, value: any) => {
    setForm(prev => ({ ...prev, [key]: value }))
  }

  const getQuote = async () => {
    const res = await fetch("/api/quote", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify(form)
    })

    const data = await res.json()
    setResult(data.data)
  }

  return (
    <div className="p-4 max-w-md mx-auto bg-white rounded-2xl shadow">
      <h2 className="text-xl font-bold mb-3">🚛 Move Calculator</h2>

      <select
        value={form.homeSize}
        onChange={e => handleChange("homeSize", e.target.value)}
        className="w-full mb-2 p-2 border rounded"
      >
        <option value="studio">Studio</option>
        <option value="1_bed">1 Bedroom</option>
        <option value="2_bed">2 Bedroom</option>
        <option value="3_bed">3 Bedroom</option>
        <option value="4_plus">4+ Bedroom</option>
      </select>

      <input
        type="number"
        placeholder="Heavy Items"
        className="w-full mb-2 p-2 border rounded"
        onChange={e => handleChange("heavyItems", Number(e.target.value))}
      />

      <label className="block">
        <input type="checkbox" onChange={e => handleChange("stairs", e.target.checked)} />
        Stairs
      </label>

      <label className="block">
        <input type="checkbox" onChange={e => handleChange("longCarry", e.target.checked)} />
        Long Carry
      </label>

      <label className="block">
        <input type="checkbox" onChange={e => handleChange("packing", e.target.checked)} />
        Packing Needed
      </label>

      <button
        onClick={getQuote}
        className="mt-3 w-full bg-orange-500 text-white p-2 rounded"
      >
        Get Estimate
      </button>

      {result && (
        <div className="mt-4 p-3 bg-gray-100 rounded">
          <p><b>Weight:</b> {result.weight} lbs</p>
          <p><b>Tier:</b> {result.tier}</p>
          <p><b>Crew:</b> {result.crew.movers} movers</p>
          <p><b>Time:</b> {result.crew.hours} hrs</p>
          <p className="text-lg font-bold">💰 ${result.price}</p>
