"use client"

import { useEffect, useState } from "react"
import { Phone } from "lucide-react"

const messages = [
  "🚛 Same-Day Moves Available",
  "⭐ 5-Star Local Movers",
  "💪 500+ Jobs Completed",
  "📞 Call Now (906) 285-9312",
]

export default function TopBanner() {
  const [index, setIndex] = useState(0)

  useEffect(() => {
    const interval = setInterval(() => {
      setIndex((prev) => (prev + 1) % messages.length)
    }, 3000)

    return () => clearInterval(interval)
  }, [])

  const current = messages[index]

  const isCall = current.includes("Call Now")

  return (
    <div className="w-full bg-gradient-to-r from-blue-900 to-blue-700 text-white text-sm md:text-base font-medium">

      <div className="max-w-7xl mx-auto px-4 py-2 flex items-center justify-center gap-2">

        {/* Animated Text */}
        <div
          key={index}
          className={`transition-all duration-500 ${
            isCall ? "animate-pulse text-yellow-300" : ""
          }`}
        >
          {current}
        </div>

        {/* Click to Call */}
        {isCall && (
          <a
            href="tel:19062859312"
            className="flex items-center gap-1 underline hover:text-yellow-200 transition"
          >
            <Phone className="w-4 h-4" />
            Tap to Call
          </a>
        )}
      </div>
    </div>
  )
}
