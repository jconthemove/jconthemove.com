Developer logic example
type LaborQuoteInput = {
  movers: number;
  minutes: number;
  serviceType: "moving" | "loading" | "furniture" | "junk";
  paymentMode: "tokens" | "split" | "cash";
};

const TOKENS_PER_MOVER_MINUTE = 500;
const CASH_PER_MOVER_HOUR = 62.5;

function calculateLaborQuote(input: LaborQuoteInput) {
  const moverMinutes = input.movers * input.minutes;
  const tokenPrice = moverMinutes * TOKENS_PER_MOVER_MINUTE;
  const cashPrice = (input.movers * input.minutes * CASH_PER_MOVER_HOUR) / 60;

  let maxTokenCoverageRatio = 1;

  if (input.minutes >= 120) {
    maxTokenCoverageRatio = 0.5;
  } else if (input.minutes >= 60) {
    maxTokenCoverageRatio = 0.75;
  }

  const tokenCap = Math.floor(tokenPrice * maxTokenCoverageRatio);

  return {
    moverMinutes,
    tokenPrice,
    cashPrice: Number(cashPrice.toFixed(2)),
    tokenCap
  };
}
