⚙️ 1. AUTO DISPATCH ENGINE (CORE)
📁 server/autoDispatch.ts
type Crew = {
  id: number;
  online: boolean;
  jobsToday: number;
  lastJobAt?: number;
  location?: { lat: number; lng: number };
};

type Job = {
  id: number;
  services: string[];
  urgency?: "low" | "normal" | "high";
  location?: { lat: number; lng: number };
  createdAt: number;
};

export function autoAssignCrew(job: Job, crewList: Crew[]) {
  // STEP 1: Filter available crew
  const available = crewList.filter(c => c.online);

  if (available.length === 0) return null;

  // STEP 2: Score each crew
  const scored = available.map(c => {
    let score = 100;

    // Fewer jobs today = higher priority
    score -= c.jobsToday * 10;

    // Recently worked? slight penalty (prevents overload)
    if (c.lastJobAt) {
      const minutesAgo = (Date.now() - c.lastJobAt) / 60000;
      if (minutesAgo < 60) score -= 15;
    }

    // Priority jobs boost score
    if (job.urgency === "high") score += 20;

    return { ...c, score };
  });

  // STEP 3: Sort best → worst
  scored.sort((a, b) => b.score - a.score);

  return scored[0];
}

📁 server/routes/quote.ts (UPDATE)
import { autoAssignCrew } from "../autoDispatch";

router.post("/quote", async (req, res) => {
  const { services, jobData } = req.body;

  const quote = calculateQuote(services, jobData);

  const job = await db.insert(jobs).values({
    services: services.join(","),
    data: jobData,
    total: quote.total,
    status: "pending",
    createdAt: Date.now(),
  }).returning();

  const crewList = await db.select().from(crew);

  const assigned = autoAssignCrew(job[0], crewList);

  if (assigned) {
    await db.update(jobs)
      .set({
        crewAssigned: assigned.id,
        status: "assigned"
      })
      .where(eq(jobs.id, job[0].id));
  }

  res.json({
    ...quote,
    assignedCrew: assigned?.name || "Searching..."
  });
});
⚖️ 3. LOAD BALANCING SYSTEM

Already partially handled above, but let’s make it explicit.

📁 server/loadBalancer.ts
export function balanceCrew(crewList: any[]) {
  return crewList.sort((a, b) => {
    return a.jobsToday - b.jobsToday;
  });
}

Use this before scoring if needed.

export function estimateStartTime(crew: any) {
  const bufferMinutes = crew.jobsToday * 90;

  return new Date(Date.now() + bufferMinutes * 60000);
}

export function calculateCrewBonus(job: any) {
  let bonus = 0;

  // High priority jobs pay more
  if (job.urgency === "high") bonus += 20;

  // Late hours bonus
  const hour = new Date().getHours();
  if (hour > 18) bonus += 15;

  // Low crew availability bonus
  if (job.availableCrew < 2) bonus += 25;

  return bonus;
}
Store it:
await db.update(jobs)
  .set({ crewBonus: bonus })
  .where(eq(jobs.id, job.id));
