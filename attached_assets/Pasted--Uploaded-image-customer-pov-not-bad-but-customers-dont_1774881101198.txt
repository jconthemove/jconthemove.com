function matchCrew(contractors, job) {
  let movers = contractors.filter(c =>
    hasCapability(c, "mover")
  )

  let drivers = contractors.filter(c =>
    hasCapability(c, "driver")
  )

  if (job.requiresTruck) {
    drivers = drivers.filter(c =>
      hasCapability(c, "truck_small") ||
      hasCapability(c, "truck_large")
    )
  }

  return {
    movers: movers.slice(0, job.requiredMovers),
    driver: drivers[0]
  }
}

app.post("/api/jobs/dispatch", async (req, res) => {
  const { jobId } = req.body

  const job = await db.query.jobs.findFirst({
    where: eq(jobs.id, jobId)
  })

  // 1. Get online contractors
  const online = await db.select().from(contractors)
    .where(eq(contractors.isOnline, true))

  // 2. Filter by capability
  const matched = online.filter(c =>
    matchesJobRequirements(c, job)
  )

  // 3. Create dispatch entries
  for (const c of matched) {
    await db.insert(jobDispatch).values({
      jobId,
      contractorId: c.id,
      status: "pending"
    })
  }

  // 4. Start timeout
  startDispatchTimer(jobId)

  res.json({ success: true, sentTo: matched.length })
})

function matchesJobRequirements(contractor, job) {
  const caps = getCapabilities(contractor.id)

  if (job.requiresDriver && !caps.includes("driver")) return false
  if (job.requiresTruck && !hasTruck(caps)) return false

  return caps.includes("mover")
}

async function assignJob(jobId) {
  const accepted = await db.select()
    .from(jobDispatch)
    .where(and(
      eq(jobDispatch.jobId, jobId),
      eq(jobDispatch.status, "accepted")
    ))

  if (accepted.length === 0) return

  const winner = accepted[0]

  await db.insert(jobAssignments).values({
    jobId,
    contractorId: winner.contractorId,
    role: "driver"
  })

  await db.update(jobs)
    .set({ status: "assigned" })
    .where(eq(jobs.id, jobId))

  // notify others
  await cancelOtherDispatches(jobId)
}

async function canTakeJob(contractorId, job) {
  const prefs = await getPreferences(contractorId)

  const weeklyCount = await countJobs(contractorId, "week")
  const monthlyCount = await countJobs(contractorId, "month")

  if (weeklyCount >= prefs.maxJobsPerWeek) return false
  if (monthlyCount >= prefs.maxJobsPerMonth) return false

  const todayHours = await getAssignedHours(contractorId, today())

  if (todayHours + job.estimatedHours > prefs.maxHoursPerDay)
    return false

  const isAvailable = await checkSchedule(contractorId, job.date)

  if (!isAvailable) return false

  return true
}

export async function dispatchJunkJob(jobId: string) {
  const job = await db.query.leads.findFirst({
    where: (l, { eq }) => eq(l.id, jobId)
  })

  const available = await db.query.users.findMany({
    where: (u, { and, eq }) =>
      and(
        eq(u.role, "employee"),
        eq(u.isAvailable, true)
      )
  })

  // prioritize drivers if needed
  let pool = available

  if (job.requiresDriver) {
    pool = pool.sort((a, b) =>
      (b.isDriver ? 1 : 0) - (a.isDriver ? 1 : 0)
    )
  }

  const selected = pool.slice(0, job.requiredMovers)

  if (selected.length === 0) return

  for (const worker of selected) {
    await db.insert(crewMembers).values({
      jobId,
      userId: worker.id
    })
  }

  await db.update(leads)
    .set({ status: "assigned" })
    .where(eq(leads.id, jobId))
}

export async function dispatchGenericJob(jobId) {
  const job = await getJob(jobId)

  const available = await getAvailableCrew(job)

  const result = assembleCrew(available, job)

  if (result.status === "fail") {
    console.log("No crew available")
    return
  }

  for (const member of result.crew) {
    await db.insert(crewMembers).values({
      jobId,
      userId: member.id,
      role: member.role
    })
  }

  await db.update(leads)
    .set({ status: "assigned" })
    .where(eq(leads.id, jobId))
}
