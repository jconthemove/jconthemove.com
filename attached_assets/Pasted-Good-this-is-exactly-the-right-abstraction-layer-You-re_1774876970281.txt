function matchCrew(contractors, job) {
  let movers = contractors.filter(c =>
    hasCapability(c, "mover")
  )

  let drivers = contractors.filter(c =>
    hasCapability(c, "driver")
  )

  if (job.requiresTruck) {
    drivers = drivers.filter(c =>
      hasCapability(c, "truck_small") ||
      hasCapability(c, "truck_large")
    )
  }

  return {
    movers: movers.slice(0, job.requiredMovers),
    driver: drivers[0]
  }
}

🟢 6. FRONTEND: LIVE CREW INDICATOR
React hook
import { useEffect, useState } from "react"

export function useCrewStatus() {
  const [count, setCount] = useState(0)

  useEffect(() => {
    const fetchStatus = async () => {
      const res = await fetch("/api/crew/online")
      const data = await res.json()
      setCount(data.total)
    }

    fetchStatus()
    const interval = setInterval(fetchStatus, 5000)

    return () => clearInterval(interval)
  }, [])

  return count
}
