app.post("/api/jobs/:id/advance", (req, res) => {
  const job = jobs.find(j => j.id == req.params.id);
  if (!job) return res.status(404).send("Not found");

  const next = req.body.nextStage;

  if (!allowedTransitions[job.stage].includes(next)) {
    return res.status(400).send("Invalid transition");
  }

  job.stage = next;
  job.history.push({ stage: next, time: new Date() });

  // TOKEN TRIGGER
  if (next === "completed") {
    job.tokens = Math.floor(job.price * 15);
  }

  if (next === "tokens_issued") {
    // simulate wallet credit
    console.log("Tokens issued:", job.tokens);
  }

  res.json(job);
});

// ================= FRONTEND =================

import { useEffect, useState } from "react";

const stages = [
  "lead_created",
  "quoted",
  "approved",
  "scheduled",
  "in_progress",
  "completed",
  "paid",
  "tokens_issued"
];

function getNextStage(current) {
  const i = stages.indexOf(current);
  return stages[i + 1];
}

export default function PipelineBoard() {
  const [jobs, setJobs] = useState([]);

  const load = () => fetch("/api/jobs").then(r => r.json()).then(setJobs);

  useEffect(() => { load(); }, []);

  const advance = async (job) => {
    const next = getNextStage(job.stage);
    if (!next) return;

    await fetch(`/api/jobs/${job.id}/advance`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ nextStage: next })
    });

    load();
  };

  return (
    <div className="p-4 space-y-4">
      <h1 className="text-xl font-bold">Job Pipeline</h1>

      {jobs.map(job => (
        <div key={job.id} className="bg-zinc-900 p-4 rounded-xl">
          <div className="flex justify-between">
            <div>
              <p className="font-bold">{job.service}</p>
              <p className="text-sm">{job.address}</p>
            </div>
            <div className="text-orange-400">{job.stage}</div>
          </div>

          <div className="mt-3 flex gap-2">
            {stages.map((s, i) => (
              <div
                key={i}
                className={`h-2 flex-1 rounded ${
                  stages.indexOf(job.stage) >= i
                    ? "bg-green-500"
                    : "bg-zinc-700"
                }`}
              />
            ))}
          </div>

          <button
            onClick={() => advance(job)}
            className="mt-3 bg-orange-500 px-3 py-1 rounded"
          >
            Advance
          </button>

          {job.tokens && (
            <p className="mt-2 text-green-400">
              Tokens: {job.tokens}
            </p>
          )}
        </div>
      ))}
    </div>
  );
}

// ================= ADVANCE STAGE =================

import { eq } from "drizzle-orm";

app.post("/api/jobs/:id/advance", async (req, res) => {
  const id = Number(req.params.id);
  const [job] = await db.select().from(jobs).where(eq(jobs.id, id));

  if (!job) return res.status(404).send("Not found");

  const next = req.body.nextStage;

  if (!allowedTransitions[job.stage]?.includes(next)) {
    return res.status(400).send("Invalid transition");
  }

  let tokens = job.tokens;

  if (next === "completed") {
    tokens = Math.floor(job.price * 15);
  }

  const updated = await db.update(jobs)
    .set({
      stage: next,
      tokens,
      history: [...(job.history || []), { stage: next, time: new Date() }],
      updatedAt: new Date()
    })
    .where(eq(jobs.id, id))
    .returning();

  res.json(updated[0]);
});

app.post("/api/jobs/:id/issue-tokens", async (req, res) => {
  const id = Number(req.params.id);

  const [job] = await db.select().from(jobs).where(eq(jobs.id, id));
  if (!job) return res.status(404).send("Not found");

  if (job.stage !== "paid") {
    return res.status(400).send("Must be paid first");
  }

  // insert ledger entry
  await db.insert(ledger).values({
    userId: job.userId,
    jobId: job.id,
    amount: job.tokens,
    type: "earn"
  });

  await db.update(jobs)
    .set({ stage: "tokens_issued" })
    .where(eq(jobs.id, id));

  res.json({ success: true });
});

// ================= REDEEM MODAL =================

function RedeemModal({ onClose }) {
  const [amount, setAmount] = useState(0);
  const [wallet, setWallet] = useState("");
  const [loading, setLoading] = useState(false);

  const submit = async () => {
    setLoading(true);

    await fetch("/api/redeem", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        amount: Number(amount),
        walletAddress: wallet
      })
    });

    setLoading(false);
    onClose();
  };

  return (
    <div className="fixed inset-0 bg-black/70 flex items-center justify-center">
      <div className="bg-zinc-900 p-4 rounded-xl w-80 space-y-3">
        <h2 className="font-bold">Redeem Tokens</h2>

        <input
          className="input"
          placeholder="Amount"
          onChange={e => setAmount(e.target.value)}
        />

        <input
          className="input"
          placeholder="Wallet Address"
          onChange={e => setWallet(e.target.value)}
        />

        <button
          onClick={submit}
          className="bg-green-500 w-full py-2 rounded"
        >
          {loading ? "Processing..." : "Submit"}
        </button>

        <button
          onClick={onClose}
          className="text-gray-400 w-full"
        >
          Cancel
        </button>
      </div>
    </div>
  );
}

app.get("/api/treasury/analytics", async (req, res) => {

  const allRedemptions = await db.select().from(redemptions);

  const earned = allLedger
    .filter(l => l.type === "earn")
    .reduce((sum, l) => sum + l.amount, 0);

  const spent = allLedger
    .filter(l => l.type === "spend")
    .reduce((sum, l) => sum + Math.abs(l.amount), 0);

  const pending = allRedemptions
    .filter(r => r.status === "pending")
    .reduce((sum, r) => sum + r.amount, 0);

  const completed = allRedemptions
    .filter(r => r.status === "sent")
    .reduce((sum, r) => sum + r.amount, 0);

  const liability = pending + completed;

  res.json({
    earned,
    spent,
    pending,
    completed,
    liability
  });
});
