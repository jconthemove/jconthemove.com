import { useEffect, useState } from "react";

export default function DispatchDashboard() {
  const [jobs, setJobs] = useState<any[]>([]);
  const [crew, setCrew] = useState<any[]>([]);

  const fetchData = async () => {
    const j = await fetch("/api/admin/jobs").then(r => r.json());
    const c = await fetch("/api/admin/crew").then(r => r.json());
    setJobs(j);
    setCrew(c);
  };

  useEffect(() => {
    fetchData();
    const interval = setInterval(fetchData, 3000); // live refresh
    return () => clearInterval(interval);
  }, []);

  const assignCrew = async (jobId: number, crewId: number) => {
    await fetch("/api/admin/assign", {
      method: "POST",
      body: JSON.stringify({ jobId, crewId }),
    });
    fetchData();
  };

  return (
    <div className="p-6 text-white bg-black min-h-screen">
      <h1 className="text-2xl font-bold mb-6">Dispatch Dashboard</h1>

      <div className="grid grid-cols-2 gap-6">
        {/* JOB BOARD */}
        <div>
          <h2 className="text-xl mb-2">Live Jobs</h2>
          {jobs.map((job) => (
            <div key={job.id} className="bg-gray-800 p-4 mb-3 rounded">
              <p>Job #{job.id}</p>
              <p>Services: {job.services}</p>
              <p>Total: ${job.total}</p>
              <p>Status: {job.status}</p>

              <select
                onChange={(e) =>
                  assignCrew(job.id, Number(e.target.value))
                }
              >
                <option>Select Crew</option>
                {crew.map((c) => (
                  <option key={c.id} value={c.id}>
                    {c.name} ({c.jobsToday} jobs)
                  </option>
                ))}
              </select>
            </div>
          ))}
        </div>

        {/* CREW STATUS */}
        <div>
          <h2 className="text-xl mb-2">Crew Status</h2>
          {crew.map((c) => (
            <div key={c.id} className="bg-gray-800 p-4 mb-3 rounded">
              <p>{c.name}</p>
              <p>Status: {c.online ? "Online" : "Offline"}</p>
              <p>Jobs Today: {c.jobsToday}</p>
            </div>
          ))}
        </div>
      </div>
    </div>
  );
}
🔌 BACKEND ROUTES
📁 server/routes/admin.ts
import express from "express";
import { db } from "../db";
import { jobs } from "../db/schema";
import { crew } from "../db/schema";

const router = express.Router();

// GET JOBS
router.get("/admin/jobs", async (req, res) => {
  const allJobs = await db.select().from(jobs);
  res.json(allJobs);
});

// GET CREW
router.get("/admin/crew", async (req, res) => {
  const allCrew = await db.select().from(crew);
  res.json(allCrew);
});

// ASSIGN CREW
router.post("/admin/assign", async (req, res) => {
  const { jobId, crewId } = req.body;

  await db.update(jobs)
    .set({ crewAssigned: crewId, status: "assigned" })
    .where(jobs.id.eq(jobId));

  await db.update(crew)
    .set({ jobsToday: crew.jobsToday + 1 })
    .where(crew.id.eq(crewId));

  res.json({ success: true });
});

export default router;
