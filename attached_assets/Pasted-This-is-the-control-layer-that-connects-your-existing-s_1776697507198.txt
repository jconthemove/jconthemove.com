🎯 OBJECTIVE

Create a single orchestration entry point:

POST /api/pipeline/run

That executes, in order:

Input → Pricing → Demand → Surge → Dispatch → Schedule → Incentives → Persist → Notify → Respond

Do not rebuild existing modules. Wrap and call them.

⚙️ CORE ENTRY
📁 server/pipeline/index.ts
import { buildContext } from "./context";
import { pricingStep } from "./steps/pricing.step";
import { demandStep } from "./steps/demand.step";
import { surgeStep } from "./steps/surge.step";
import { dispatchStep } from "./steps/dispatch.step";
import { scheduleStep } from "./steps/schedule.step";
import { incentivesStep } from "./steps/incentives.step";
import { persistStep } from "./steps/persist.step";
import { notifyStep } from "./steps/notify.step";

export async function runPipeline(input) {
  const ctx = await buildContext(input);

  await pricingStep(ctx);
  await demandStep(ctx);
  await surgeStep(ctx);
  await dispatchStep(ctx);
  await scheduleStep(ctx);
  await incentivesStep(ctx);
  await persistStep(ctx);
  await notifyStep(ctx);

  return {
    jobId: ctx.job.id,
    total: ctx.total,
    multiplier: ctx.multiplier,
    crewAssigned: ctx.crew,
    eta: ctx.eta,
    upsells: ctx.upsells
  };
}

export async function buildContext(input) {

    total: 0,
    multiplier: 1,
    demandScore: 0,
    crew: null,
    eta: null,
    bonus: 0,
    upsells: [],
    job: null,
    now: Date.now()
  };
}

🧩 STEP IMPLEMENTATIONS
1) Pricing
export async function pricingStep(ctx) {
  const quote = await getQuote(ctx.input);
  ctx.quote = quote;
  ctx.total = quote.total;
}
2) Demand
export function demandStep(ctx) {
  const { demandScore } = predictDemand(ctx.input);
  ctx.demandScore = demandScore;
}
3) Surge
export function surgeStep(ctx) {
  const res = applySurge(ctx.total, ctx.demandScore, ctx.input.availableCrew || 0);
  ctx.total = res.total;
  ctx.multiplier = res.multiplier;
}
4) Dispatch
export async function dispatchStep(ctx) {
  const crewList = await getCrew();
  const crew = selectCrew(ctx.input, crewList);
  ctx.crew = crew;
}
5) Schedule
export function scheduleStep(ctx) {
  ctx.eta = estimateStartTime(ctx.crew);
}
6) Incentives
export function incentivesStep(ctx) {
  ctx.bonus = calculateCrewBonus(ctx.input);
}
7) Persist
export async function persistStep(ctx) {
  const job = await createJob({
    ...ctx.input,
    total: ctx.total,
    crew: ctx.crew?.id,
    eta: ctx.eta,
    bonus: ctx.bonus
  });

  ctx.job = job;
}
8) Notify
export function notifyStep(ctx) {
  if (ctx.crew) sendSMS(ctx.job);
}
🌐 ROUTE

import express from "express";

const router = express.Router();

router.post("/pipeline/run", async (req, res) => {

  res.json(result);
});

export default router;
