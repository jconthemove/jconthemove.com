"use client"

import { useState } from "react"
import JunkCards from "./JunkCards"
// (you will plug these in next)
const MovingEstimator = () => <div className="p-4">Moving Flow Coming</div>
const LaborQuick = () => <div className="p-4">Labor Flow Coming</div>
const SnowFlow = () => <div className="p-4">Snow Flow Coming</div>

export default function ServiceSelector() {
  const [selected, setSelected] = useState<string | null>(null)

  const services = [
    { id: "moving", label: "🚛 Moving Help", sub: "Full moves & loading" },
    { id: "junk", label: "🧹 Junk Removal", sub: "Fast pickup & disposal" },
    { id: "labor", label: "💪 Labor Only", sub: "Hourly help" },
    { id: "snow", label: "❄️ Snow Removal", sub: "Seasonal services" }
  ]

  return (
    <div className="w-full max-w-md mx-auto">

      {/* SERVICE BUTTONS */}
      <div className="grid grid-cols-2 gap-3 p-4">
        {services.map(s => (
          <button
            key={s.id}
            onClick={() => setSelected(s.id)}
            className={`p-4 rounded-xl border text-left transition-all duration-200 ${
              selected === s.id
                ? "border-orange-500 bg-orange-500/10 scale-[1.03]"
                : "border-neutral-800 bg-neutral-900"
            }`}
          >
            <div className="font-semibold">{s.label}</div>
            <div className="text-xs text-gray-400 mt-1">{s.sub}</div>
          </button>
        ))}
      </div>

      {/* DYNAMIC PANEL */}
      <div className="px-4 pb-10">
        <div className="transition-all duration-300 ease-in-out">

          {!selected && (
            <div className="text-center text-gray-400 mt-6">
              Select a service to continue
            </div>
          )}

          {selected === "junk" && <JunkCards />}
          {selected === "moving" && <MovingEstimator />}
          {selected === "labor" && <LaborQuick />}
          {selected === "snow" && <SnowFlow />}

        </div>
      </div>
    </div>
  )
}
