➕ 4. ADD-ONS SYSTEM (WHERE YOU MAKE MONEY)
/components/AddOns.tsx
"use client"

import { useState } from "react"

export default function AddOns({ basePrice }) {
  const [items, setItems] = useState({
    mattress: 0,
    fridge: 0,
    gym: 0
  })

  function update(type, value) {
    setItems(prev => ({
      ...prev,
      [type]: Math.max(0, prev[type] + value)
    }))
  }

  const total =
    basePrice +
    items.mattress * 50 +
    items.fridge * 100 +
    items.gym * 100

  return (
    <div className="mt-4 p-4 bg-neutral-900 rounded-xl">
      <h3 className="font-semibold mb-2">➕ Add Items</h3>

      {["mattress", "fridge", "gym"].map(type => (
        <div key={type} className="flex justify-between items-center mb-2">
          <span className="capitalize">{type}</span>

          <div className="flex items-center gap-2">
            <button onClick={() => update(type, -1)}>-</button>
            <span>{items[type]}</span>
            <button onClick={() => update(type, 1)}>+</button>
          </div>
        </div>
      ))}

      <div className="mt-4 border-t border-neutral-700 pt-3">
        <p className="text-lg font-bold">
          Total: ${total}
        </p>

        <button className="w-full mt-3 bg-orange-500 py-3 rounded-xl font-semibold">
          Book Now
        </button>
      </div>
    </div>
  )
}
