🧠 1) DEMAND PREDICTION ENGINE

Start simple: use your own job history.

📁 server/prediction.ts
type Job = {
  createdAt: number;
  lat: number;
  lng: number;
};

export function predictDemand(jobs: Job[]) {
  const now = new Date();
  const hour = now.getHours();

  // Count jobs in last 7 days by hour
  const buckets: Record<number, number> = {};

  jobs.forEach(j => {
    const d = new Date(j.createdAt);
    const h = d.getHours();
    buckets[h] = (buckets[h] || 0) + 1;
  });

  const currentDemand = buckets[hour] || 0;

  // Normalize into 0–1 score
  const max = Math.max(...Object.values(buckets), 1);
  const demandScore = currentDemand / max;

  return {
    hour,
    demandScore, // 0 (dead) → 1 (peak)
  };
}

export function applySurge(baseTotal: number, demandScore: number, crewAvailable: number) {
  let multiplier = 1;

  // Demand-based surge
  if (demandScore > 0.7) multiplier += 0.2;
  if (demandScore > 0.9) multiplier += 0.4;

  // Low crew = higher price
  if (crewAvailable <= 1) multiplier += 0.3;

  // Too many idle crews = discount
  if (crewAvailable > 5 && demandScore < 0.3) multiplier -= 0.15;

  return {
    total: Math.round(baseTotal * multiplier),
    multiplier
  };
}

export function suggestCrewPosition(demandZones: any[], crew: any) {
  if (!crew.online) return null;

  // Find highest demand zone
  const bestZone = demandZones.sort((a, b) => b.score - a.score)[0];

  return {
    targetLat: bestZone.lat,
    targetLng: bestZone.lng,
    reason: "High demand zone"
  };
}

import { haversine } from "./geo";

export function chooseBestCrew(job: any, crewList: any[], demandScore: number) {
  const available = crewList.filter(c => c.online && c.lat && c.lng);

  const scored = available.map(c => {
    const distance = haversine(job, c);

    let score = 100;

    // Distance penalty
    score -= distance * 8;

    // Load balance
    score -= c.jobsToday * 6;

    // High-value jobs → prioritize best crews
    if (job.total > 800) score += 15;

    // High demand → speed matters more
    if (demandScore > 0.7) score -= distance * 5;

    return { ...c, score };
  });

  scored.sort((a, b) => b.score - a.score);

  return scored[0];
}
